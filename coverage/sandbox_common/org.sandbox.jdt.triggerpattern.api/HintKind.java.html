package org.sandbox.jdt.triggerpattern.api;

/**
 * Kind of hint, matching NetBeans hint model.
 * 
 * @since 1.2.2
 */
public enum HintKind {
	/**
	 * Inspection hint - automatically detects and reports issues.
	 */
	INSPECTION,
	
	/**
	 * Action hint - user-triggered transformation (e.g., refactoring).
	 */
	ACTION
}
